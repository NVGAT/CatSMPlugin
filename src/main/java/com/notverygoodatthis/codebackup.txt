package com.notverygoodatthis;

import org.bukkit.*;
import org.bukkit.command.Command;
import org.bukkit.command.CommandExecutor;
import org.bukkit.command.CommandSender;
import org.bukkit.entity.Player;

public class CommandKit implements CommandExecutor {
    @Override
    public boolean onCommand(CommandSender sender, Command command, String label, String[] args) {
        if(sender instanceof Player) {
            Player playerSender = (Player) sender;
            if(label.length() != 0) {
                Player target = Bukkit.getPlayerExact(args[0]);
                if(target == null) {
                    playerSender.sendMessage("Player " + target.getDisplayName() + " is not online.");
                }
                else {
                    if(playerSender.getInventory().getItemInMainHand().getType() == Material.PLAYER_HEAD) {
                        if(target.getStatistic(Statistic.DEATHS) > SMPlugin.MAX_LIVES - 1) {
                            playerSender.getInventory().remove(Material.PLAYER_HEAD);
                            Bukkit.getBanList(BanList.Type.NAME).pardon(target.getDisplayName());
                            target.setStatistic(Statistic.DEATHS, 0);
                        }
                        else {
                            playerSender.sendMessage(target.getDisplayName() + " is alive and does not need a revival.");
                        }
                    }
                    else {
                        playerSender.sendMessage("Hold a player head in your hand to revive!");
                    }
                }
            }
        }
        return true;
    }
}
